class TCGCost(APIModel):
    """TCG cost."""

    element: str = Aliased("cost_type")
    value: int = Aliased("cost_value")

    def __fix_element(cls, value: str) -> str:

            "CostTypeDendro": "Dendro",
            "CostTypeElectro": "Electro",
            "CostTypeGeo": "Geo",

            "CostTypePyro": "Pyro",
            "CostTypeAnemo": "Anemo",
            "CostTypeSame": "Same",
            "CostTypeVoid": "Void",
